### Real-time Announcements

```typescript
// WebSocket integration for live updates
const useRealTimeUpdates = () => {
  const [socket, setSocket] = useState<WebSocket | null>(null);
  const [announcements, setAnnouncements] = useState<Announcement[]>([]);

  useEffect(() => {
    const ws = new WebSocket(process.env.NEXT_PUBLIC_WS_URL!);
    
    ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      if (data.type === 'announcement') {
        setAnnouncements(prev => [data.announcement, ...prev]);
      }
    };

    setSocket(ws);
    return () => ws.close();
  }, []);

  return { announcements, socket };
};
```

### User Authentication & Roles

```typescript
// Auth.js configuration with role-based access
export const authOptions: NextAuthOptions = {
  providers: [
    GoogleProvider({
      clientId: process.env.GOOGLE_CLIENT_ID!,
      clientSecret: process.env.GOOGLE_CLIENT_SECRET!,
    }),
  ],
  callbacks: {
    session: ({ session, token }) => ({
      ...session,
      user: {
        ...session.user,
        id: token.sub,
        role: token.role as 'admin' | 'organizer' | 'participant',
      },
    }),
    jwt: ({ token, user }) => {
      if (user) {
        token.role = user.role;
      }
      return token;
    },
  },
};
```

```typescript
const EventRegistration: React.FC<{ event: Event }> = ({ event }) => {
  const [formData, setFormData] = useState({
    teamName: '',
    teamSize: 1,
    contactEmail: '',
    requirements: ''
  });

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    
    const registration = {
      eventId: event.id,
      ...formData,
      registeredAt: new Date()
    };

    await registerForEvent(registration);
  };

  return (
    <form onSubmit={handleSubmit} className="registration-form">
      <input
        type="text"
        placeholder="Team Name"
        value={formData.teamName}
        onChange={(e) => setFormData(prev => ({ ...prev, teamName: e.target.value }))}
        required
      />
      {/* Additional form fields */}
      <button type="submit" disabled={!event.registrationOpen}>
        {event.registrationOpen ? 'Register Now' : 'Registration Closed'}
      </button>
    </form>
  );
};
```
