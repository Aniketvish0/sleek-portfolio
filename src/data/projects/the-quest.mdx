## Overview

**The Quest** is a personal challenge tracking system designed to monitor and motivate the completion of three ambitious goals within 6 months: solving 500 DSA problems, earning ₹300,000, and maintaining fitness consistency. The platform features real-time countdown, progress visualization, and motivational design to maintain momentum throughout the journey.

## Technical Implementation

### Countdown System

```typescript
interface QuestTimer {
  days: number;
  hours: number;
  minutes: number;
  seconds: number;
  milliseconds: number;
}

const QuestCountdown: React.FC = () => {
  const [timeLeft, setTimeLeft] = useState<QuestTimer>({
    days: 0, hours: 0, minutes: 0, seconds: 0, milliseconds: 0
  });

  useEffect(() => {
    const targetDate = new Date('2025-06-30T23:59:59'); // 6 months from start

    const interval = setInterval(() => {
      const now = new Date();
      const difference = targetDate.getTime() - now.getTime();

      if (difference > 0) {
        setTimeLeft({
          days: Math.floor(difference / (1000 * 60 * 60 * 24)),
          hours: Math.floor((difference % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60)),
          minutes: Math.floor((difference % (1000 * 60 * 60)) / (1000 * 60)),
          seconds: Math.floor((difference % (1000 * 60)) / 1000),
          milliseconds: difference % 1000
        });
      }
    }, 10); // Update every 10ms for smooth animation

    return () => clearInterval(interval);
  }, []);

  return (
    <div className="countdown-display text-6xl font-bold text-center">
      {timeLeft.days} days, {timeLeft.hours} hours, {timeLeft.minutes} minutes, {timeLeft.seconds} seconds, {timeLeft.milliseconds} ms
    </div>
  );
};
```

### Progress Tracking System

```typescript
interface QuestProgress {
  dsa: {
    current: number;
    target: number;
    percentage: number;
  };
  money: {
    current: number;
    target: number;
    percentage: number;
  };
  fitness: {
    current: number;
    target: number;
    percentage: number;
  };
}

const ProgressTracker: React.FC = () => {
  const [progress, setProgress] = useState<QuestProgress>({
    dsa: { current: 0, target: 500, percentage: 0 },
    money: { current: 0, target: 300000, percentage: 0 },
    fitness: { current: 5, target: 177, percentage: 2.8 }
  });

  const updateProgress = (category: keyof QuestProgress, value: number) => {
    setProgress(prev => ({
      ...prev,
      [category]: {
        ...prev[category],
        current: value,
        percentage: (value / prev[category].target) * 100
      }
    }));

    // Save to localStorage for persistence
    localStorage.setItem('quest-progress', JSON.stringify(progress));
  };

  return (
    <div className="grid grid-cols-1 md:grid-cols-3 gap-6">
      {Object.entries(progress).map(([key, data]) => (
        <ProgressCard
          key={key}
          title={key.toUpperCase()}
          current={data.current}
          target={data.target}
          percentage={data.percentage}
          onUpdate={(value) => updateProgress(key as keyof QuestProgress, value)}
        />
      ))}
    </div>
  );
};
```

### Progress Card Component

```typescript
interface ProgressCardProps {
  title: string;
  current: number;
  target: number;
  percentage: number;
  onUpdate: (value: number) => void;
}

const ProgressCard: React.FC<ProgressCardProps> = ({
  title, current, target, percentage, onUpdate
}) => {
  return (
    <motion.div
      className="bg-gradient-to-br from-purple-600 to-blue-600 p-6 rounded-xl text-white"
      whileHover={{ scale: 1.05 }}
      transition={{ type: "spring", stiffness: 300 }}
    >
      <h3 className="text-2xl font-bold mb-4">{title}</h3>

      {/* Progress Bar */}
      <div className="w-full bg-white/20 rounded-full h-4 mb-4">
        <motion.div
          className="bg-white rounded-full h-4"
          initial={{ width: 0 }}
          animate={{ width: `${percentage}%` }}
          transition={{ duration: 1, ease: "easeOut" }}
        />
      </div>

      {/* Statistics */}
      <div className="text-center">
        <p className="text-3xl font-bold">
          {title === 'MONEY' ? `₹${current.toLocaleString()}` : current}
        </p>
        <p className="text-lg opacity-80">
          {title === 'MONEY' ? `₹${target.toLocaleString()}` : target} {getUnit(title)}
        </p>
        <p className="text-sm opacity-60">{percentage.toFixed(1)}% Complete</p>
      </div>

      {/* Update Button */}
      <motion.button
        className="w-full mt-4 bg-white/20 hover:bg-white/30 rounded-lg p-2"
        whileTap={{ scale: 0.95 }}
        onClick={() => {
          const newValue = prompt(`Update ${title} progress:`);
          if (newValue && !isNaN(Number(newValue))) {
            onUpdate(Number(newValue));
          }
        }}
      >
        Update Progress
      </motion.button>
    </motion.div>
  );
};
```

## Progress Persistence

### **Local Storage Management**

```typescript
class QuestStorage {
  private static KEY = 'quest-progress';

  static save(progress: QuestProgress): void {
    localStorage.setItem(
      this.KEY,
      JSON.stringify({
        ...progress,
        lastUpdated: new Date().toISOString(),
      }),
    );
  }

  static load(): QuestProgress | null {
    const stored = localStorage.getItem(this.KEY);
    if (!stored) return null;

    try {
      return JSON.parse(stored);
    } catch {
      return null;
    }
  }

  static clear(): void {
    localStorage.removeItem(this.KEY);
  }
}
```
