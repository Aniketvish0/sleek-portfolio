## Technical Implementation

### Festival Website Architecture

```typescript
interface FestivalEvent {
  id: string;
  title: string;
  category: 'cultural' | 'academic' | 'creative' | 'robotics' | 'management' | 'technical' | 'gaming' | 'sports';
  organizer: string;
  status: 'open' | 'closed' | 'upcoming';
  description: string;
  registrationLink: string;
  image: string;
}

const FestivalPage: React.FC = () => {
  const [events, setEvents] = useState<FestivalEvent[]>([]);
  const [selectedCategory, setSelectedCategory] = useState<string>('all');
  const [countdown, setCountdown] = useState<TimeLeft>({});

  useEffect(() => {
    // Countdown to festival start
    const interval = setInterval(() => {
      const festivalDate = new Date('2025-03-15');
      const now = new Date();
      const difference = festivalDate.getTime() - now.getTime();

      setCountdown({
        days: Math.floor(difference / (1000 * 60 * 60 * 24)),
        hours: Math.floor((difference % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60)),
        minutes: Math.floor((difference % (1000 * 60 * 60)) / (1000 * 60)),
        seconds: Math.floor((difference % (1000 * 60)) / 1000)
      });
    }, 1000);

    return () => clearInterval(interval);
  }, []);

  return (
    <div className="festival-website">
      <HeroSection countdown={countdown} />
      <AboutSection />
      <EventsSection events={events} category={selectedCategory} />
      <MemoriesGallery />
      <SponsorsSection />
    </div>
  );
};
```

const EventsSection: React.FC<{ events: FestivalEvent[]; category: string }> = ({ events, category }) => {

### Memories Gallery with Hover Effects

```typescript
const MemoriesGallery: React.FC = () => {
  const [hoveredIndex, setHoveredIndex] = useState<number | null>(null);

  const memories = Array.from({ length: 16 }, (_, i) => ({
    id: i,
    image: `/memories/moonstone-2k24-${i}.jpg`,
    title: `Moonstone 2K24 Memory ${i + 1}`
  }));

  return (
    <section className="memories-section py-16 bg-black text-white">
      <div className="container mx-auto px-4">
        <h2 className="text-4xl font-bold text-center mb-4">
          HOVER/TAP TO SEE MEMORIES
        </h2>
        <h3 className="text-2xl text-center mb-12">MOONSTONE 2K24 MEMORIES</h3>

        <div className="grid grid-cols-4 md:grid-cols-8 gap-2">
          {memories.map((memory, index) => (
            <motion.div
              key={memory.id}
              className="relative aspect-square overflow-hidden rounded-lg cursor-pointer"
              onHoverStart={() => setHoveredIndex(index)}
              onHoverEnd={() => setHoveredIndex(null)}
              whileHover={{ scale: 1.1, zIndex: 10 }}
              transition={{ duration: 0.3 }}
            >
              <Image
                src={memory.image}
                alt={memory.title}
                fill
                className="object-cover grayscale hover:grayscale-0 transition-all duration-300"
              />
              {hoveredIndex === index && (
                <motion.div
                  className="absolute inset-0 bg-black/70 flex items-center justify-center"
                  initial={{ opacity: 0 }}
                  animate={{ opacity: 1 }}
                >
                  <p className="text-white text-sm text-center p-2">{memory.title}</p>
                </motion.div>
              )}
            </motion.div>
          ))}
        </div>
      </div>
    </section>
  );
};
```

```typescript
const EventRegistration: React.FC<{ eventId: string }> = ({ eventId }) => {
  const [registration, setRegistration] = useState({
    name: '',
    email: '',
    phone: '',
    college: '',
    year: ''
  });

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();

    try {
      const response = await fetch('/api/register', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ eventId, ...registration })
      });

      if (response.ok) {
        toast.success('Registration successful! Check email for confirmation.');
      }
    } catch (error) {
      toast.error('Registration failed. Please try again.');
    }
  };

  return (
    <form onSubmit={handleSubmit} className="registration-form">
      {/* Form fields */}
      <button type="submit" className="register-btn">
        Register Now
      </button>
    </form>
  );
};
```
