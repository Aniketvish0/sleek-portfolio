### Voting System

```typescript
const VotingSystem: React.FC<{ startup: Startup }> = ({ startup }) => {
  const [hasVoted, setHasVoted] = useState(false);
  const [voteCount, setVoteCount] = useState(startup.votes);
  const { data: session } = useSession();

  const handleVote = async () => {
    if (!session) {
      toast.error('Please login to vote');
      return;
    }

    if (hasVoted) {
      toast.info('You have already voted for this startup');
      return;
    }

    try {
      const response = await fetch(`/api/startups/${startup.id}/vote`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ userId: session.user.id })
      });

      if (response.ok) {
        setVoteCount(prev => prev + 1);
        setHasVoted(true);
        toast.success('Vote recorded!');
      }
    } catch (error) {
      toast.error('Failed to record vote');
    }
  };

  return (
    <div className="voting-section">
      <button
        onClick={handleVote}
        className={`vote-btn ${hasVoted ? 'voted' : 'available'}`}
        disabled={hasVoted}
      >
        üëç {voteCount}
      </button>
      <p className="vote-count">{voteCount} votes</p>
    </div>
  );
};
```

```typescript
const StartupSubmissionForm: React.FC = () => {
  const [formData, setFormData] = useState({
    title: '',
    description: '',
    category: 'technology',
    tags: '',
    problem: '',
    solution: '',
    targetMarket: '',
    businessModel: ''
  });

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();

    const startupData = {
      ...formData,
      tags: formData.tags.split(',').map(tag => tag.trim()),
      founderId: session?.user?.id
    };

    await submitStartup(startupData);
  };

  return (
    <motion.form
      onSubmit={handleSubmit}
      className="submission-form bg-white rounded-lg shadow-lg p-8"
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
    >
      <h3 className="text-2xl font-bold mb-6">Submit Your Startup Idea</h3>

      <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
        <div>
          <label className="block text-sm font-medium mb-2">Startup Title</label>
          <input
            type="text"
            value={formData.title}
            onChange={(e) => setFormData(prev => ({ ...prev, title: e.target.value }))}
```

```typescript
const IdeaValidation: React.FC<{ startup: Startup }> = ({ startup }) => {
  const [feedback, setFeedback] = useState('');
  const [rating, setRating] = useState(0);

  const submitFeedback = async () => {
    const feedbackData = {
      startupId: startup.id,
      rating,
      comment: feedback,
      userId: session?.user?.id

      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(feedbackData)
    });
  };

  return (
    <div className="idea-validation mt-6">
      <h4 className="text-lg font-semibold mb-4">Provide Feedback</h4>

      <div className="rating-section mb-4">

        <div className="flex space-x-1">
          {[1, 2, 3, 4, 5].map((star) => (
            <button
              key={star}
              onClick={() => setRating(star)}
              className={`text-2xl ${star <= rating ? 'text-yellow-400' : 'text-gray-300'}`}
            >
              ‚≠ê
            </button>
          ))}
        </div>
      </div>

      <textarea
        value={feedback}
        onChange={(e) => setFeedback(e.target.value)}
```
